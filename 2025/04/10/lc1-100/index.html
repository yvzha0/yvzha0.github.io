1. 两数之和

解题步骤：哈希表法

  1. 初始化哈希表：创建一个哈希表来存储数组元素的值及其对应的索引。
  2. 遍历数组：逐个访问数组中的元素，索引为i，当前元素值为a = nums[i]。
  3. 计算目标差值：对于每个元素a，计算需要查找的差值b = target - a。
  4. 查找差值是否存在：在哈希表中查找b：
    • 存在：返回当前索引i和哈希表中存储的b的索引。
    • 不存在：将当前元素a及其索引i存入哈希表，继续检查下一个元素。
#include <vector>
#include <unordered_map>

using namespace std;

class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        unordered_map<int,int> umap;
        for(int i=0;i<nums.size();i++){
            int a = nums[i];
            int b = target - a;
            if(umap.find(b) != umap.end()){
                return {i,umap[b]};
            }
            umap[a] = i;
        }
        return {};
    }
};
import java.util.HashMap;
import java.util.Map;

class Solution {
    public int[] twoSum(int[] nums, int target) {
        int[] result = new int[2];
        Map<Integer, Integer> map = new HashMap<>();
        int length = nums.length;
        for(int i=0;i<length;i++){
            int a = nums[i];
            int b = target - a;
            if(map.containsKey(b)){
                result[0] = i;
                result[1] = map.get(b);
                return result;
            }
            map.put(a,i);
        }
        return result;
    }
}

2. 两数相加

解题步骤：模拟法

  1. 初始化指针和进位变量：
    • cur1、cur2：用于遍历链表 l1 和 l2。
    • cur：指向当前结果链表的最后一个有效节点。
    • jinwei：用于记录加法过程中的进位。
  2. 同时遍历两个链表：
    • 当 cur1 和 cur2 都不为空时，逐位相加。
    • 把结果更新到 cur1->val 上，即直接修改 l1 链表。
    • 记录进位 jinwei。
    • 如果即将到达末尾，则用 cur 记录当前节点，为后续追加节点做准备。
  3. 处理 l1 剩余部分（如果 l1 更长）：
    • 如果还有进位，则继续加上当前节点值，并更新。
    • 否则直接跳出。
  4. 处理 l2 剩余部分（如果 l2 更长）：
    • 如果还有进位，创建新节点追加到结果链表尾部。
    • 如果没有进位了，就把 l2 剩下的节点接到结果链表上（复用原链表节点，节省空间）。
  5. 处理最后的进位（如果还有）：
    • 创建一个新节点存储最后的进位，追加到结果链表末尾。最后返回l1。
#include "ListNode.h"
class Solution {
public:
    ListNode* addTwoNumbers(ListNode* l1, ListNode* l2) {
        ListNode* cur1 = l1;
        ListNode* cur2 = l2;
        ListNode* cur;
        int jinwei = 0;
        while(cur1 != nullptr && cur2 != nullptr){
            int sum = cur1->val + cur2->val + jinwei;
            cur1->val = sum % 10;
            jinwei = sum / 10;
            if(cur1->next == nullptr || cur2->next == nullptr){
                cur = cur1;
            }
            cur1 = cur1->next;
            cur2 = cur2->next;
        }
        while(cur1 != nullptr){
            if(jinwei != 0){
                int sum = cur1->val + jinwei;
                cur1->val = sum % 10;
                jinwei = sum / 10;
            }else{
                break;
            }
            if(cur1->next == nullptr){
                cur = cur1;
            }
            cur1 = cur1->next;
        }
        while(cur2 != nullptr){
            if(jinwei != 0){
                int sum = cur2->val + jinwei;
                auto* a = new ListNode(sum % 10);
                cur->next = a;
                cur = cur->next;
                jinwei = sum / 10;
            }else{
                cur->next = cur2;
                break;
            }
            cur2 = cur2->next;
        }
        if(jinwei != 0){
            auto* a = new ListNode(jinwei);
            cur->next = a;
        }
        return l1;
    }
};

		ListNode cur1 = l1;
		ListNode cur2 = l2;
		ListNode pre = null;
		int jinwei = 0;
		while(cur1 != null && cur2 != null){
			int sum = cur1.val + cur2.val + jinwei;
			cur1.val = sum % 10;
			jinwei = sum / 10;
			if(cur1.next == null){
				pre = cur1;
			}
			cur1 = cur1.next;
			cur2 = cur2.next;
		}
		while(cur1 != null){
			if(jinwei != 0){
				int sum = cur1.val + jinwei;
				cur1.val = sum % 10;
				jinwei = sum / 10;
				if(cur1.next == null){
					pre = cur1;
				}
				cur1 = cur1.next;
			}else{
				break;
			}
		}
		while(cur2 != null){
			if(jinwei != 0){
				int sum = cur2.val + jinwei;
				cur2.val = sum % 10;
				jinwei = sum / 10;
				pre.next = cur2;
				pre = pre.next;
				cur2 = cur2.next;
			}else{
				pre.next = cur2;
				break;
			}
		}
		if(jinwei != 0){
            pre.next = new ListNode(jinwei);
		}
        return l1;
    }
}

3. 无重复字符的最长子串

解题步骤：动态规划法

  1. 初始化辅助数组：
    • 创建一个 dp 数组用于动态规划，dp[i] 表示以第 i 个字符为结尾的最长不重复子串的长度。
    • 初始化 dp[0] = 1，即第一个字符的最长子串长度为 1。
    • 初始化结果 ans = 1，用于记录最大值。
  2. 边界情况处理：如果字符串长度为 0，直接返回 0。
  3. 遍历字符串，更新 dp 数组：
    • 从第二个字符开始，依次向前遍历过去的 dp[i-1] 个字符（即前一个不重复子串的长度范围内）。
    • 如果在这个范围内出现了与 s[i] 相同的字符：
      • 说明有重复，更新 dp[i] = i - j，即从重复字符之后开始的新子串长度。
      • 设置 flag = false 表示出现重复。
    • 如果没有重复字符：
      • 可以直接在前一个子串的基础上延长一位，dp[i] = dp[i-1] + 1。
  4. 更新最大长度：每次计算完 dp[i] 后，更新 ans = max(ans, dp[i])。
  5. 返回最终结果：返回最长不重复子串的长度 ans。
#include <string>

using namespace std;

class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        int length = s.size();
        if(length == 0){
            return 0;
        }
        int dp[length];
        dp[0] = 1;
        int ans = 1;
        for(int i=1;i<length;i++){
            bool flag = true;
            for(int j=i-1;j>=i-dp[i-1];j--){
                if(s[i] == s[j]){
                    dp[i] = i - j;
                    flag = false;
                    break;
                }
            }
            if (flag){
                dp[i] = dp[i-1] + 1;
            }
            if(dp[i] > ans){
                ans = dp[i];
            }
        }
        return ans;
    }
};
class Solution {
    public int lengthOfLongestSubstring(String s) {
        int length = s.length();
        if(length == 0 || length == 1){
            return length;
        }
        int[] dp = new int[length];
        dp[0] = 1;
        int ans = 1;
        for(int i=1;i<length;i++){
            boolean flag = true;
            for(int j=i-1;j>=i-dp[i-1];j--){
                if(s.charAt(i) == s.charAt(j)){
                    dp[i] = i - j;
                    flag = false;
                    break;
                }
            }
            if(flag){
                dp[i] = dp[i-1] + 1;
            }
            if(dp[i] > ans){
                ans = dp[i];
            }
        }
        return ans;
    }
}

4. 寻找两个正序数组的中位数

解题步骤：二分查找法

  1. 中位数转为找第 k 小元素的问题：
    • 奇数长度：中位数是第 (m + n + 1)/2 个。
    • 偶数长度：中位数是第 (m+n)/2 和 (m+n)/2 + 1 个数的平均值。
  2. 使用二分查找来找第 k 小元素：
    • 比较 nums1[k/2 - 1] 和 nums2[k/2 - 1]。
    • 较小的一侧可以排除掉前 k/2 个元素。
    • 不断更新 k、起始索引直到找到第 k 小的元素。
  3. 终止条件处理：
    • 如果某个数组已经遍历完了，直接在另一个数组找。
    • 如果 k == 1，取两个当前起始位置最小的元素返回。
#include <vector>

using namespace std;

class Solution {
public:
    double findMedianSortedArrays(vector<int>& nums1, vector<int>& nums2) {
        int total = nums1.size() + nums2.size();
        if (total % 2 == 1) {
            return getKthElement(nums1, nums2, total / 2 + 1);
        } else {
            return (getKthElement(nums1, nums2, total / 2) +
                    getKthElement(nums1, nums2, total / 2 + 1)) / 2.0;
        }
    }

private:
    static int getKthElement(const vector<int>& nums1, const vector<int>& nums2, int k) {
        int index1 = 0, index2 = 0;
        int m = nums1.size(), n = nums2.size();

        while (true) {
            if (index1 == m) {
                return nums2[index2 + k - 1];
            }
            if (index2 == n) {
                return nums1[index1 + k - 1];
            }
            if (k == 1) {
                return min(nums1[index1], nums2[index2]);
            }

            int newIndex1 = min(index1 + k / 2 - 1, m - 1);
            int newIndex2 = min(index2 + k / 2 - 1, n - 1);

            int pivot1 = nums1[newIndex1];
            int pivot2 = nums2[newIndex2];

            if (pivot1 <= pivot2) {
                k -= (newIndex1 - index1 + 1);
                index1 = newIndex1 + 1;
            } else {
                k -= (newIndex2 - index2 + 1);
                index2 = newIndex2 + 1;
            }
        }
    }
};

class Solution {
    public double findMedianSortedArrays(int[] nums1, int[] nums2) {
        int length = nums1.length + nums2.length;
		if(length % 2 == 1){
			return findPos(nums1, nums2 ,length / 2 + 1);
		}
		return (double) (findPos(nums1, nums2, length / 2) + findPos(nums1, nums2, length / 2 + 1)) / 2;
    }

	private int findPos(int[] nums1, int[] nums2, int pos){
		int index1 = 0;
		int index2 = 0;
		int m = nums1.length;
		int n = nums2.length;
		while (true) {
			if (index1 == m) {
				return nums2[index2 + pos - 1];
			}
			if (index2 == n) {
				return nums1[index1 + pos - 1];
			}
			if (pos == 1) {
				return Integer.min(nums1[index1], nums2[index2]);
			}
			int newIndex1 = Integer.min(index1 + pos / 2 - 1, m-1);
			int newIndex2 = Integer.min(index2 + pos / 2 - 1, n-1);
			if (nums1[newIndex1] <= nums2[newIndex2]) {
				pos -= (newIndex1 - index1 + 1);
				index1 = newIndex1 + 1;
			}else{
				pos -= (newIndex2 - index2 + 1);
				index2 = newIndex2 + 1;
			}
		}
	}
}

5. 最长回文子串

解题步骤：中心扩展法

  1. 初始化答案字符串：
    • 如果字符串为空或长度为 1，直接返回；
    • 初始化结果 ans 为第一个字符（确保至少有一个回文子串）。
  2. 遍历字符串，尝试以每个字符为中心向两边扩展（奇数长度的回文）：
    • 设当前字符下标为 i，左右指针分别为 j = i - 1 和 k = i + 1；
    • 当左右指针在范围内，且 s[j] == s[k]，说明是回文，继续扩展；
    • 若扩展后回文长度 count 大于当前最长 ans，更新 ans 为新的回文子串。
  3. 遍历字符串，尝试以每对相邻字符为中心扩展（偶数长度的回文）：
    • 若相邻两个字符 s[i] == s[i+1]，从这对字符出发进行中心扩展；
    • 左右指针设置为 j = i - 1 和 k = i + 2，以同样的方式向外扩展；
    • 更新最长回文子串 ans。
  4. 返回最终的最长回文子串 ans。
#include <string>

using namespace std;

class Solution {
public:
    string longestPalindrome(string s) {
        int length = s.size();
        if(length == 0){
            return "";
        }
        if(length == 1){
            return s;
        }
        string ans;
        ans.push_back(s[0]);
        for(int i=1;i<length;i++){
            int count = 1;
            int j = i - 1;
            int k = i + 1;
            while(j >= 0 && k < length){
                if(s[j] == s[k]){
                    count += 2;
                    j--;
                    k++;
                }else{
                    break;
                }
            }
            if(count > ans.size()){
                ans = s.substr(j+1,count);
            }
        }
        for(int i=0;i<length-1;i++){
            int count;
            if(s[i] == s[i+1]){
                count = 2;
            }else{
                continue;
            }
            int j = i - 1;
            int k = i + 2;
            while(j >= 0 && k < length){
                if(s[j] == s[k]){
                    count += 2;
                    j--;
                    k++;
                }else{
                    break;
                }
            }
            if(count > ans.size()){
                ans = s.substr(j+1,count);
            }
        }
        return ans;
    }
};
class Solution {
    public String longestPalindrome(String s) {
		int length = s.length();
		if(length == 1){
			return s;
		}
		String ans = "";
		ans += s.charAt(0);
		for(int i=1;i<length;i++){
			int j = i - 1;
			int k = i + 1;
			int count = 1;
			while(j >= 0 && k < length){
				if(s.charAt(j) == s.charAt(k)){
					count += 2;
					j--;
					k++;
				}else{
					break;
				}
			}
			if(count > ans.length()){
				ans = s.substring(j+1,k);
			}
		}
		for(int i=0;i<length-1;i++){
			int count;
			if(s.charAt(i) == s.charAt(i+1)){
				count = 2;
			}else{
				continue;
			}
			int j = i - 1;
			int k = i + 2;
			while(j >= 0 && k < length){
				if(s.charAt(j) == s.charAt(k)){
					count += 2;
					j--;
					k++;
				}else{
					break;
				}
			}
			if(count > ans.length()){
				ans = s.substring(j+1,k);
			}
		}
        return ans;
    }
}

6. Z 字形变换

解题步骤：按行模拟

  1. 处理特殊情况：如果 numRows == 1，说明不需要 Z 字形排列，直接返回原字符串。

  2. 定义变量：

    • length: 字符串的长度
    • ans: 最终的结果字符串
    • Z 字形一轮的周期为 cycleLen = 2 * numRows - 2，即从最顶层往下，再折回到最顶层所需的字符数。

  3. 处理每一行的字符：

    a. 第一行（行号 = 0）

    • 这行的字符位置是间隔 cycleLen 的，比如索引：0，0 + cycleLen，0 + 2*cycleLen，…
    • 把这些字符依次加入结果中。

    b. 中间行（行号 = 1 到 numRows - 2）

    • 这些行的字符每轮出现两次，分别在下降和上升阶段。
    • 对于第 i 行（1 ≤ i < numRows - 1）：
      • 第一次出现在位置 j = i；
      • 第二次出现在 j + cycleLen - 2 * i；
      • 然后继续循环下一个周期 j += cycleLen；
      • 每轮取两个字符（如果还在字符串范围内）。

    c. 最后一行（行号 = numRows - 1）

    • 和第一行类似，索引是：numRows - 1, numRows - 1 + cycleLen, …
    • 依次加入结果中。

  4. 返回结果字符串 ans。

#include <string>

using namespace  std;

class Solution {
public:
    string convert(string s, int numRows) {
        if(numRows == 1){
            return s;
        }
        int length = s.size();
        string ans;
        for(int i=0;i<length;i+=2*numRows-2){
            ans.push_back(s[i]);
        }
        for(int i=1;i<numRows-1;i++){
            for(int j=i;j<length;){
                ans.push_back(s[j]);
                j += 2 * numRows - 2 * (i + 1);
                if(j < length){
                    ans.push_back(s[j]);
                }
                j += 2 * i;
            }
        }
        for(int i=numRows-1;i<length;i+=2*numRows-2){
            ans.push_back(s[i]);
        }
        return ans;
    }
};
class Solution {
    public String convert(String s, int numRows) {
        int length = s.length();
		if(numRows == 1 || numRows == length){
			return s;
		}
		StringBuilder ans = new StringBuilder();
		for(int i=0;i<length;i+=2*numRows-2){
			ans.append(s.charAt(i));
		}
		for(int i=1;i<numRows-1;i++){
			for(int j=i;j<length;){
				ans.append(s.charAt(j));
				j += 2 * (numRows - i - 1);
				if(j < length){
					ans.append(s.charAt(j));
				}
				j += 2 * i;
			}
		}
		for(int i=numRows-1;i<length;i+=2*numRows-2){
			ans.append(s.charAt(i));
		}
		return ans.toString();
    }
}

7. 整数反转

解题步骤：模拟

  1. 初始化变量：定义一个整数变量 ans = 0，用于保存反转后的结果。

  2. 循环处理每一位数字：

    • 每次从末尾取一位数字 digit = x % 10。
    • 把当前这位数字追加到 ans 的末尾（ans = ans * 10 + digit）。
    • 去掉 x 的末尾数字：x /= 10。

  3. 在追加前判断是否溢出：

    因为不能用 64 位整数，要提前判断乘以 10 之后是否会超出 int 范围：

    • 正溢出判断：

      if (ans > INT_MAX / 10 || (ans == INT_MAX / 10 && digit > 7)) return 0;
      • 因为 INT_MAX = 2147483647，所以当 ans == 214748364 且下一位大于 7 时会溢出。

    • 负溢出判断：

      if (ans < INT_MIN / 10 || (ans == INT_MIN / 10 && digit < -8)) return 0;
      • INT_MIN = -2147483648，所以当 ans == -214748364 且下一位小于 -8 时会溢出。

  4. 循环结束后返回反转后的 ans。

#include <climits>

class Solution {
public:
    int reverse(int x) {
        int ans = 0;
        while (x != 0) {
            int digit = x % 10;
            x /= 10;

            if (ans > INT_MAX / 10 || (ans == INT_MAX / 10 && digit > 7)) return 0;
            if (ans < INT_MIN / 10 || (ans == INT_MIN / 10 && digit < -8)) return 0;

            ans = ans * 10 + digit;
        }
        return ans;
    }
};

class Solution {
    public int reverse(int x) {
        int ans = 0;
		while(x != 0){
			int digit = x % 10;
			x /= 10;
			if(ans > Integer.MAX_VALUE / 10){
				return 0;
			}
			if(ans < Integer.MIN_VALUE / 10){
				return 0;
			}
			ans = ans * 10 + digit;
		}
		return ans;
    }
}

8. 字符串转换整数 (atoi)

解题步骤：模拟

  1. 跳过前导空格：从字符串开头开始，忽略所有空格字符，直到遇到第一个非空格字符。
  2. 判断符号：检查当前字符是否为 '-' 或 '+'，用于确定结果的正负。如果存在符号，记录下来，并将索引向后移动。
  3. 逐位转换数字：
    • 从当前位置开始遍历，只要当前字符是数字字符（’0’-‘9’），就将其转换为对应数值并累加到结果中（即每次乘以 10 加上新数位）。
    • 同时使用一个 long 类型的变量存储结果，以便及时判断是否超出 32 位整数范围。
  4. 处理溢出：
    • 如果累加后的值超过 INT_MAX（2³¹ − 1）或小于 INT_MIN（−2³¹），则立即返回边界值。
  5. 返回结果：
    • 最后根据之前记录的符号，返回最终结果（若为负则取反）。
#include "string"
#include <cmath>
using namespace std;

class Solution {
public:
    int myAtoi(string s) {
        int i = 0;
        int len = s.size();
        bool isZ = true;
        int max = INT_MAX;
        int min = INT_MIN;
        long res = 0;
        while (s[i] == ' ') {
            i++;
        }
        if (s[i] == '-') {
            isZ = false;
            i++;
        }else if(s[i] == '+'){
            i++;
        }
        while (s[i] >= '0' && s[i] <= '9' && i < len) {
            res = res * 10 + s[i] - '0';
            if (isZ) {
                if (res >= max) {
                    res = max;
                    return int(res);
                }
            } else {
                if (-res <= min) {
                    res = min;
                    return int(res);
                }
            }
            i++;
        }
        return isZ ? int(res) : (-res);
    }
};

9. 回文数

解题步骤：双指针法

  1. 排除负数情况：因为负数的符号 - 使其不可能是回文，所以如果 x < 0，直接返回 false。
  2. 计算整数的位数（位数计数）：用一个临时变量 a（初始为 x）不断除以 10，直到变成 0，同时计数器 weishu 自增，用来记录整数 x 的位数。
  3. 初始化两个指针：
    • 指针 i 表示从高位开始的索引，初始为 0。
    • 指针 j 表示从低位开始的索引，初始为 weishu（虽然 j 在后续判断中只控制循环次数，并不直接取位）。
  4. 逐位比较高位和低位：在循环中：
    • 使用 int(x / pow(10, weishu - i - 1)) % 10 取出当前的高位第 i 位数字（从左往右）。
    • 使用 a % 10 取出当前最低位的数字（从右往左）。
    • 比较两个数字是否相等，不相等则返回 false。
    • 否则更新 a /= 10，移除最低位数字。
    • 增加 i，减少比较次数，直到比较到中间。
  5. 全部位匹配，返回 true
#include <cmath>

class Solution {
public:
    bool isPalindrome(int x) {
        if(x < 0){
            return false;
        }
        int a = x;
        int weishu = 0;
        while(a != 0){
            a /= 10;
            weishu++;
        }
        a = x;
        int i = 0,j = weishu;
        while(i < j){
            int b = int(x / pow(10,weishu-i-1)) % 10;
            int c = a % 10;
            if(b != c){
                return false;
            }
            a /= 10;
            i++;
            j--;
        }
        return true;
    }
};

11. 盛最多水的容器

解题步骤：双指针法

  1. 初始化双指针：设置两个指针 left 和 right，分别指向数组的起始和末尾位置，表示当前考虑的两条垂线。
  2. 计算当前面积：
    • 在每一步中，计算两个指针间的宽度 b = right - left。
    • 取两个高度中较小值 a = min(height[left], height[right]) 作为有效高度。
    • 当前容器能盛的水量为 a * b，更新最大值记录。
  3. 移动指针优化结果：
    • 为了尽可能找到更大的面积，移动较短的那一侧的指针（因为只有提高短边才可能增大面积）：
      • 若左侧高度较小，则移动左指针 left++；
      • 若右侧高度较小或相等，则移动右指针 right--。
  4. 重复判断：重复上述步骤直到两个指针相遇。
  5. 返回最大面积：循环结束后返回记录的最大水量。
#include <vector>

using namespace std;

class Solution {
public:
    int maxArea(vector<int>& height) {
        int left = 0;
        int right = height.size()-1;
        int ans = 0;
        while(left < right){
            int b = right - left;
            int a;
            if(height[left] < height[right]){
                a = height[left];
                left++;
            }else{
                a = height[right];
                right--;
            }
            int captin = a * b;
            ans = max(ans,captin);
        }
        return ans;
    }
};

13. 罗马数字转整数

解题步骤

    • i 用于遍历字符串 s。
  3. 遍历字符串（到倒数第二个字符）：
    • 对于当前字符 s[i] 和下一个字符 s[i+1]：
      • 如果 s[i] < s[i+1]，说明是“减法规则”，如 IV, IX, XL 等：
        • 将 umap[s[i+1]] - umap[s[i]] 加入总和。
        • 跳过下一位（i++）。
      • 否则：
        • 将 umap[s[i]] 加入总和。
  4. 检查是否还有最后一个字符未处理：如果 i == length - 1，说明最后一个字符没参与“减法规则”，单独加上它的值。
  5. 返回结果 ans。

#include <unordered_map>

class Solution {
public:
    unordered_map<char,int> umap;
    int romanToInt(string s) {
        umap['I'] = 1;umap['V'] = 5;umap['X'] = 10;umap['L'] = 50;umap['C'] = 100;umap['D'] = 500;umap['M'] = 1000;
        int length = s.size();
        int ans = 0;
        int i = 0;
        for(i;i<length-1;i++){
            if(umap[s[i]] < umap[s[i+1]]){
                ans += umap[s[i+1]] - umap[s[i]];
                i++;
            }else{
                ans += umap[s[i]];
            }
        }
        if(i == length - 1){
            ans += umap[s[i]];
        }
        return ans;
    }
};

15. 三数之和

解题步骤：排序+双指针

  1. 排序数组：对输入数组 nums 进行升序排序，便于后续使用双指针方法并去除重复三元组。
  2. 固定第一个数：
    • 从数组起始位置开始遍历，用变量 first 表示三元组的第一个元素索引。
    • 如果当前 nums[first] 与前一个相同，跳过以避免重复解。
    • 如果 nums[first] > 0，由于数组已排序，后续所有数都为正，三数之和不可能为 0，提前结束遍历。
  3. 使用双指针查找其余两个数：
    • 对于每个 first，从 first + 1 开始遍历 second，并设置 third 为数组末尾索引。
    • 若 nums[second] == nums[second - 1]（且 second > first+1），则跳过以避免重复。
    • 移动 third 指针，直到 nums[first] + nums[second] + nums[third] <= 0。
    • 若此时三数之和等于 0，说明找到一个符合条件的三元组，添加到结果中。
  4. 终止条件：当 second 与 third 相遇时，说明当前 first 所能组合的三元组已全部处理。
  5. 返回结果：返回收集到的所有不重复的三元组。
#include <vector>
#include <algorithm>

using namespace std;

class Solution {
public:
    vector<vector<int>> threeSum(vector<int>& nums) {
        sort(nums.begin(), nums.end());
        vector<vector<int>> ans;
        int len = nums.size();
        for(int first=0;first<len-2;first++){
            if(first > 0 && nums[first] == nums[first-1]){
                continue;
            }
            if(nums[first] > 0){
                break;
            }
            int third = len - 1;
            for(int second=first+1;second<len-1;second++){
                if(second > first + 1 && nums[second] == nums[second-1]){
                    continue;
                }
                while(third > second && nums[first] + nums[second] + nums[third] > 0){
                    third--;
                }
                if(third == second){
                    break;
                }
                if(nums[first] + nums[second] + nums[third] == 0){
                    ans.push_back({nums[first], nums[second], nums[third]});
                }
            }
        }
        return ans;
    }
};

16. 最接近的三数之和

解题步骤：排序+双指针

  1. 数组排序：先将数组 nums 升序排序，以便后续使用双指针技术优化查找。
  2. 初始化最优解：定义变量 ans 存储当前找到的最接近目标值的三数之和，初始值设置为一个足够大的整数。
  3. 遍历第一个元素：
    • 使用变量 first 遍历数组的前 n - 2 个位置。
    • 可跳过重复值（nums[first] == nums[first - 1]）优化效率，但非必要。
  4. 双指针寻找剩余两个数：
    • 固定 first 后，从 first + 1 到末尾，用 second 向右移动，third 向左移动。
    • 对于每对 (second, third)，计算当前三数之和 sum = nums[first] + nums[second] + nums[third]。
  5. 比较并更新最优解：
    • 若 sum 恰好等于 target，直接返回结果；
    • 否则判断当前 sum 与 target 的距离是否比 ans 更接近，若是则更新 ans。
  6. 移动指针以尝试更优解：
    • 若当前 sum > target，说明总和偏大，移动 third 向左；
    • 若 sum < target，说明总和偏小，移动 second 向右。
  7. 返回最终答案：所有可能组合遍历后，返回记录的最接近目标的三数之和 ans。
#include <vector>
#include <algorithm>

using namespace std;

class Solution {
public:
    int threeSumClosest(vector<int>& nums, int target) {
        int len = nums.size();
        sort(nums.begin(), nums.end());
        int ans = INT_MAX - 1e4;
        for(int first = 0;first<len-2;first++){
            if(first > 0 && nums[first] == nums[first-1]){
                continue;
            }
            int third = len - 1;
            for(int second=first+1;second<len-1;second++){
                while(third > second && nums[first] + nums[second] + nums[third] > target){
                    if(nums[first] + nums[second] + nums[third] - target < abs(ans - target)){
                        ans = nums[first] + nums[second] + nums[third];
                    }
                    third--;
                }
                if(second == third){
                    break;
                }
                if(nums[first] + nums[second] + nums[third] == target){
                    return target;
                }
                if(-(nums[first] + nums[second] + nums[third] - target) < abs(ans - target)){
                    ans = nums[first] + nums[second] + nums[third];
                }
            }
        }
        return ans;
    }
};

21. 合并两个有序链表

解题步骤

  1. 定义两个指针 cur1 和 cur2：

    • cur2 指向第二个链表的当前节点，初始为 list2。
  2. 创建虚拟头结点：
    • 新建一个值为 0 的节点 res 作为新链表的起点。
    • 指针 cur 用于在新链表中移动，初始指向 res。
  3. 循环合并两个链表：
    • 当 cur1 和 cur2 都不为空时，比较它们的值：
      • 如果 cur1->val <= cur2->val，将 cur1 接到新链表末尾，移动 cur1。
      • 否则，将 cur2 接到新链表末尾，移动 cur2。
    • 每次都更新 cur 到当前新链表末尾。
  4. 处理剩余节点：如果一个链表遍历完了，把另一个链表剩下的部分直接接到新链表后面
  5. 返回结果：虚拟头结点不是真正数据的一部分，最终返回的是 res->next，即合并后的链表头指针。
#include "ListNode.h"

class Solution {
public:
    ListNode* mergeTwoLists(ListNode* list1, ListNode* list2) {
        ListNode* cur1 = list1;
        ListNode* cur2 = list2;
        ListNode* res = new ListNode(0);
        ListNode* cur = res;
        while(cur1 != nullptr && cur2 != nullptr){
            if(cur1->val <= cur2->val){
                cur->next = cur1;
                cur = cur1;
                cur1 = cur1->next;
            }else{
                cur->next = cur2;
                cur = cur2;
                cur2 = cur2->next;
            }
        }
        cur->next = cur1 == nullptr?cur2:cur1;
        return res->next;
    }
};

28. 找出字符串中第一个匹配项的下标

解题步骤：KMP算法

  1. 获取字符串长度：首先获取主字符串（haystack）和模式字符串（needle）的长度，用于后续遍历和比较。
  2. 构建前缀数组（next 数组）
    • 创建一个数组用于记录模式串中每个位置的“最长相等前后缀”的长度。
    • 遍历模式串中的每个字符，比较当前位置与之前的位置是否构成相等前后缀。
    • 如果构成，记录长度；否则，根据前一个位置回退，直到找到可以匹配的位置或回退到底。
  3. 使用 KMP 算法在主串中查找匹配位置
    • 同时遍历主串和模式串中的字符。
    • 如果字符匹配，继续比较下一个字符；如果不匹配，则根据前缀数组跳回合适的位置继续匹配。
    • 一旦模式串完全匹配成功，就返回主串中第一个匹配位置的下标。
  4. 返回匹配结果
    • 如果在主串中找到了完整的模式串，返回匹配起始位置的下标。
    • 如果遍历完主串都没有找到匹配项，则返回 -1。
#include <string>
#include "vector"

using namespace std;

class Solution {
public:
    int strStr(string haystack, string needle) {
        int len1 = haystack.size();
        int len2 = needle.size();
        vector<int> next(len2,0);
        int j = 0;
        for(int i=1;i<len2;i++){
            while(j > 0 && needle[i] != needle[j]){
                j = next[j-1];
            }
            if(needle[i] == needle[j]){
                j++;
            }
            next[i] = j;
        }
        j = 0;
        for(int i=0;i<len1;i++){
            while (j > 0 && haystack[i] != needle[j]) {
                j = next[j - 1];
            }
            if(haystack[i] == needle[j]){
                j++;
            }
            if(j == len2){
                return i - len2 + 1;
            }
        }
        return -1;
    }
};

35. 搜索插入位置35. 搜索插入位置

解题步骤：二分查找

  1. 初始化左右边界：设置两个指针 left 和 right，分别指向数组的起始位置和末尾位置。
  2. 二分查找：进入 while 循环，条件是 left <= right，在每次循环中执行以下操作：
    • 计算中间索引 mid，即 (left + right) / 2。
  3. 目标值匹配：
    • 如果 nums[mid] == target，则说明目标值已经找到了，直接返回该索引 mid。
  4. 目标值大于中间元素：
    • 如果 nums[mid] < target，说明目标值应该在右半部分，更新 left = mid + 1。
    • 检查 mid 是否为数组的最后一个元素，或者下一个元素是否大于目标值。如果满足条件，说明目标值应该插入到 mid + 1 位置，返回该位置。
  5. 目标值小于中间元素：
    • 如果 nums[mid] > target，说明目标值应该在左半部分，更新 right = mid - 1。
    • 检查 mid 是否为数组的第一个元素，或者前一个元素是否小于目标值。如果满足条件，说明目标值应该插入到 mid 位置，返回该位置。
#include <vector>

using namespace std;

class Solution {
public:
    int searchInsert(vector<int>& nums, int target) {
        int left = 0;
        int right = nums.size()-1;
        while(left <= right){
            int mid = (left + right) / 2;
            if(nums[mid] == target){
                return mid;
            }else if(nums[mid] < target){
                if(mid == nums.size() - 1 || nums[mid+1] > target){
                    return mid + 1;
                }
                left = mid + 1;
            }else{
                if(mid == 0 || nums[mid-1] < target){
                    return mid;
                }
                right = mid - 1;
            }
        }
        return 0;
    }
};

58. 最后一个单词的长度

解题步骤

  1. 去除末尾空格：首先，计算字符串 s 的长度，并将 length 设置为字符串的最后一个字符的索引。接着，通过一个 while 循环跳过字符串末尾的所有空格字符。
  2. 遍历字符串：从字符串的开始遍历直到去除末尾空格后的长度位置。初始化一个变量 ans 为 0，用来记录当前单词的长度。
  3. 统计单词的长度：
    • 遍历过程中，如果遇到空格字符（' '），则表示当前单词结束，将 ans 重置为 0。
    • 如果遇到字母字符，则将 ans 增加 1，继续计算当前单词的长度。
  4. 返回结果：遍历结束后，ans 将保存最后一个单词的长度，最终返回该值。
#include <string>

using namespace std;

class Solution {
public:
    int lengthOfLastWord(string s) {
        int length = s.size() - 1;
        while(s[length] == ' '){
            length--;
        }
        int ans = 0;
        for(int i=0;i<=length;i++){
            if(s[i] == ' '){
                ans = 0;
                continue;
            }
            ans++;
        }
        return ans;
    }
};

66. 加一

解题步骤

  1. 从最低位开始加一：对数组的最后一个数字加一，并计算该位的值（取模 10）和进位（除以 10）。
  2. 处理进位传播：从倒数第二位开始向前遍历数组：
    • 如果没有进位了（即进位为 0），则直接返回结果。
    • 如果有进位，就将当前位与进位相加，更新当前位的值和新的进位。
  3. 处理最高位的进位：遍历结束后如果仍然存在进位，说明需要在数组最前面插入一个新数字（通常是 1）。
  4. 返回结果数组：最后返回更新后的数组，即为原整数加一后的结果。
#include <vector>

using namespace std;

class Solution {
public:
    vector<int> plusOne(vector<int>& digits) {
        int length = digits.size();
        int jw = 0;
        int sum = digits[length-1] + 1;
        digits[length-1] = sum % 10;
        jw = sum / 10;
        for(int i=length-2;i>=0;i--){
            if(jw == 0){
                return digits;
            }
            sum = digits[i] + jw;
            digits[i] = sum % 10;
            jw = sum / 10;
        }
        if(jw != 0){
            digits.insert(digits.begin(),jw);
        }
        return digits;
    }
};

67. 二进制求和

  1. 初始化指针和进位：设置两个指针分别指向两个字符串的末尾（最低位），并初始化一个变量表示进位为 0。
  2. 从最低位开始逐位相加：
    • 同时遍历两个字符串的有效位，计算当前位置两个数字与进位之和。
    • 将当前位的结果（sum % 2）转为字符加入结果字符串中。
    • 更新进位为 sum / 2。
  3. 处理剩余位数：
    • 若某一个字符串还有剩余位，继续处理这些位并加入进位。
    • 每次计算当前位与进位之和，更新该位和进位。
  4. 处理最终进位：遍历结束后，如果进位不为 0，则将进位加入结果中。
  5. 反转结果字符串：由于是从低位开始相加的，最终需要将结果字符串反转后返回。
  6. 返回结果：返回得到的二进制和字符串。
#include <string>
#include <algorithm>

using namespace std;

class Solution {
public:
    string addBinary(string a, string b) {
        int i = a.size() - 1;
        int j = b.size() - 1;
        int jw = 0;
        string ans;
        while(i >= 0 && j >= 0){
            int sum = a[i] - '0' + b[j] - '0' + jw;
            ans.push_back(sum % 2 + '0');
            jw  = sum / 2;
            i--;
            j--;
        }
        while(i >= 0){
            int sum = a[i] - '0' + jw;
            ans.push_back(sum % 2 + '0');
            jw = sum / 2;
            i--;
        }
        while(j >= 0){
            int sum = b[j] - '0' + jw;
            ans.push_back(sum % 2 + '0');
            jw = sum / 2;
            j--;
        }
        if(jw != 0){
            ans.push_back(jw + '0');
        }
        reverse(ans.begin(),ans.end());
        return ans;
    }
};
