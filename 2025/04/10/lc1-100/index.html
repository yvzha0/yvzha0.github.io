1. 两数之和

  1. 初始化哈希表：创建一个哈希表来存储数组元素的值及其对应的索引。
  2. 遍历数组：逐个访问数组中的元素，索引为i，当前元素值为a = nums[i]。
  3. 计算目标差值：对于每个元素a，计算需要查找的差值b = target - a。
  4. 查找差值是否存在：在哈希表中查找b：
    • 存在：返回当前索引i和哈希表中存储的b的索引。
    • 不存在：将当前元素a及其索引i存入哈希表，继续检查下一个元素。
#include <vector>
#include <unordered_map>

using namespace std;

class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        unordered_map<int,int> umap;
        for(int i=0;i<nums.size();i++){
            int a = nums[i];
            int b = target - a;
            if(umap.find(b) != umap.end()){
                return {i,umap[b]};
            }
            umap[a] = i;
        }
        return {};
    }
};

2. 两数相加

  1. 初始化指针和进位变量：
    • cur1、cur2：用于遍历链表 l1 和 l2。
    • cur：指向当前结果链表的最后一个有效节点。
    • jinwei：用于记录加法过程中的进位。
  2. 同时遍历两个链表：
    • 当 cur1 和 cur2 都不为空时，逐位相加。
    • 把结果更新到 cur1->val 上，即直接修改 l1 链表。
    • 记录进位 jinwei。
    • 如果即将到达末尾，则用 cur 记录当前节点，为后续追加节点做准备。
  3. 处理 l1 剩余部分（如果 l1 更长）：
    • 如果还有进位，则继续加上当前节点值，并更新。
    • 否则直接跳出。
  4. 处理 l2 剩余部分（如果 l2 更长）：
    • 如果还有进位，创建新节点追加到结果链表尾部。
    • 如果没有进位了，就把 l2 剩下的节点接到结果链表上（复用原链表节点，节省空间）。
  5. 处理最后的进位（如果还有）：
    • 创建一个新节点存储最后的进位，追加到结果链表末尾。最后返回l1。
#include "ListNode.h"
class Solution {
public:
    ListNode* addTwoNumbers(ListNode* l1, ListNode* l2) {
        ListNode* cur1 = l1;
        ListNode* cur2 = l2;
        ListNode* cur;
        int jinwei = 0;
        while(cur1 != nullptr && cur2 != nullptr){
            int sum = cur1->val + cur2->val + jinwei;
            cur1->val = sum % 10;
            jinwei = sum / 10;
            if(cur1->next == nullptr || cur2->next == nullptr){
                cur = cur1;
            }
            cur1 = cur1->next;
            cur2 = cur2->next;
        }
        while(cur1 != nullptr){
            if(jinwei != 0){
                int sum = cur1->val + jinwei;
                cur1->val = sum % 10;
                jinwei = sum / 10;
            }else{
                break;
            }
            if(cur1->next == nullptr){
                cur = cur1;
            }
            cur1 = cur1->next;
        }
        while(cur2 != nullptr){
            if(jinwei != 0){
                int sum = cur2->val + jinwei;
                auto* a = new ListNode(sum % 10);
                cur->next = a;
                cur = cur->next;
                jinwei = sum / 10;
            }else{
                cur->next = cur2;
                break;
            }
            cur2 = cur2->next;
        }
        if(jinwei != 0){
            auto* a = new ListNode(jinwei);
            cur->next = a;
        }
        return l1;
    }
};

3. 无重复字符的最长子串

本题考察动态规划

  1. 初始化辅助数组：
    • 创建一个 dp 数组用于动态规划，dp[i] 表示以第 i 个字符为结尾的最长不重复子串的长度。
    • 初始化 dp[0] = 1，即第一个字符的最长子串长度为 1。
    • 初始化结果 ans = 1，用于记录最大值。
  2. 边界情况处理：如果字符串长度为 0，直接返回 0。
  3. 遍历字符串，更新 dp 数组：
    • 从第二个字符开始，依次向前遍历过去的 dp[i-1] 个字符（即前一个不重复子串的长度范围内）。
    • 如果在这个范围内出现了与 s[i] 相同的字符：
      • 说明有重复，更新 dp[i] = i - j，即从重复字符之后开始的新子串长度。
      • 设置 flag = false 表示出现重复。
    • 如果没有重复字符：
      • 可以直接在前一个子串的基础上延长一位，dp[i] = dp[i-1] + 1。
  4. 更新最大长度：每次计算完 dp[i] 后，更新 ans = max(ans, dp[i])。
  5. 返回最终结果：返回最长不重复子串的长度 ans。
#include <string>

using namespace std;

class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        int length = s.size();
        if(length == 0){
            return 0;
        }
        int dp[length];
        dp[0] = 1;
        int ans = 1;
        for(int i=1;i<length;i++){
            bool flag = true;
            for(int j=i-1;j>=i-dp[i-1];j--){
                if(s[i] == s[j]){
                    dp[i] = i - j;
                    flag = false;
                    break;
                }
            }
            if (flag){
                dp[i] = dp[i-1] + 1;
            }
            if(dp[i] > ans){
                ans = dp[i];
            }
        }
        return ans;
    }
};

4. 寻找两个正序数组的中位数

  1. 中位数转为找第 k 小元素的问题：
    • 奇数长度：中位数是第 (m + n + 1)/2 个。
    • 偶数长度：中位数是第 (m+n)/2 和 (m+n)/2 + 1 个数的平均值。
  2. 使用二分查找来找第 k 小元素：
    • 比较 nums1[k/2 - 1] 和 nums2[k/2 - 1]。
    • 较小的一侧可以排除掉前 k/2 个元素。
    • 不断更新 k、起始索引直到找到第 k 小的元素。
  3. 终止条件处理：
    • 如果某个数组已经遍历完了，直接在另一个数组找。
    • 如果 k == 1，取两个当前起始位置最小的元素返回。
#include <vector>

using namespace std;

class Solution {
public:
    double findMedianSortedArrays(vector<int>& nums1, vector<int>& nums2) {
        int total = nums1.size() + nums2.size();
        if (total % 2 == 1) {
            return getKthElement(nums1, nums2, total / 2 + 1);
        } else {
            return (getKthElement(nums1, nums2, total / 2) +
                    getKthElement(nums1, nums2, total / 2 + 1)) / 2.0;
        }
    }

private:
    static int getKthElement(const vector<int>& nums1, const vector<int>& nums2, int k) {
        int index1 = 0, index2 = 0;
        int m = nums1.size(), n = nums2.size();

        while (true) {
            if (index1 == m) {
                return nums2[index2 + k - 1];
            }
            if (index2 == n) {
                return nums1[index1 + k - 1];
            }
            if (k == 1) {
                return min(nums1[index1], nums2[index2]);
            }

            int newIndex1 = min(index1 + k / 2 - 1, m - 1);
            int newIndex2 = min(index2 + k / 2 - 1, n - 1);

            int pivot1 = nums1[newIndex1];
            int pivot2 = nums2[newIndex2];

            if (pivot1 <= pivot2) {
                k -= (newIndex1 - index1 + 1);
                index1 = newIndex1 + 1;
            } else {
                k -= (newIndex2 - index2 + 1);
                index2 = newIndex2 + 1;
            }
        }
    }
};
