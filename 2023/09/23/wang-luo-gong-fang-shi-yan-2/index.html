实验2：端口扫描实验

实验要求

使用Python（Scapy库）编写端口扫描程序，对目标IP（包含IP地址段）进行扫描，完成以下功能：

1）使用ICMP协议探测主机是否开启；

2）对本机（关闭防火墙）的开放端口和非开放端口完成半连接、ACK、FIN、Null、Xmas、Windows扫描，并与Nmap扫描结果进行比较。

3）对远程（有防火墙）主机的开放端口和非开放端口完成半连接、ACK、FIN、Null、Xmas、Windows扫描，并与2）进行比较，分析结果。

实验环境

IntelliJ IDEA 2022.2.5

python 3.11.4

实验步骤及结果

  1. 在课上完成ping函数实现使用ICMP探测主机是否开启

    def ping(host):
    package = IP(dst=host) / ICMP() / b'Welcome to my world!'
    reply = sr1(package, timeout=2, verbose=False)
    if reply:
        print(f'主机{host}可达')
    else:
        print(f'主机{host}不可达')

    使用了Scapy库来构造和发送ICMP数据包，有回应就是开启的，否则是关闭的。

    结果：

  2. 模仿scan_T函数完成scan_S,scan_F,scan_X,scan_N函数，实现半连接，FIN，X_mas，Null扫描

    scan_T：

    def scan_T(dst_ip, dst_port):
    tcp_connect_scan_resp = sr1(IP(dst=dst_ip) / TCP(dport=dst_port, flags="S"), timeout=10)
    if (tcp_connect_scan_resp is None):
        print("Closed")
    elif (tcp_connect_scan_resp.haslayer(TCP)):
        print(tcp_connect_scan_resp.getlayer(TCP).flags)
        if (tcp_connect_scan_resp.getlayer(TCP).flags == "SA"):
            # 第三次握手，这句实际没有用，已经判定端口处于开放状态
            send_rst = sr(IP(dst=dst_ip) / TCP(dport=dst_port, flags="AR"),
                          timeout=1)  # 全连接 AR => ACK+RST
            print("Open")
        else:
            print(tcp_connect_scan_resp.getlayer(TCP).flags)
            print("Closed")
    else:
        print("Closed")

    首先尝试建立TCP连接，如果连接成功，它会检查TCP标志位。如果标志位为”SA”，表示第三次握手已经完成，端口处于开放状态。然后发送一个全连接请求（SYN+ACK+RST），并打印”Open”。如果连接失败或者标志位不是”SA”，则打印”Closed”。

    scan_S：

    首先尝试建立TCP连接，如果连接成功，它会检查TCP标志位。如果标志位为”SA”，表示三次握手已经完成，端口处于开放状态。然后打印”Open”。如果连接失败或者标志位不是”SA”，则打印”Closed”。

    scan_F：

    def scan_F(ip, port):
    fin_connect_scan_resp = sr1(IP(dst=ip) / TCP(dport=port, flags="F"), timeout=10)
    if fin_connect_scan_resp is None:
        print("Open|Filtered")
    elif fin_connect_scan_resp.haslayer(TCP):
        print(fin_connect_scan_resp.getlayer(TCP).flags)
        if fin_connect_scan_resp.getlayer(TCP).flags == "RA":
            print("Closed")
    else:
        print("Closed")

    发送标识位带有FIN包，返回空的话可能是开放或被防火墙过滤，如果返回的标志位是RA说明是关闭的。

    scan_X：

    发送标识位带有PSH,FIN,URG包，返回空的话可能是开放或被防火墙过滤，如果返回的标志位是RA说明是关闭的。

    scan_N：

    def scan_N(ip, port):
    Null_connect_scan_resp = sr1(IP(dst=ip) / TCP(dport=port, flags=''), timeout=10)

    发送标识位是空的包，返回空的话可能是开放或被防火墙过滤，如果返回的标志位是RA说明是关闭的。

实验结果

扫描老师的主机的开放和未开放端口，开放的全部成功，未开放的端口用FIN，Xmas，Null扫描结果是开放的，可能被过滤了

扫描本机虚拟机开放和未开放端口结果全部正确

最终代码

from scapy.all import *
from scapy.layers.inet import IP, TCP, ICMP

'''
通过设置flags位为'*',根据不同的扫描方式修改‘*’的值
'''

def scan_T(dst_ip, dst_port):
    tcp_connect_scan_resp = sr1(IP(dst=dst_ip) / TCP(dport=dst_port, flags="S"), timeout=10)
    if (tcp_connect_scan_resp is None):
        print(f"端口{dst_port}Closed")
    elif (tcp_connect_scan_resp.haslayer(TCP)):
        print(tcp_connect_scan_resp.getlayer(TCP).flags)
        if (tcp_connect_scan_resp.getlayer(TCP).flags == "SA"):
            # 第三次握手，这句实际没有用，已经判定端口处于开放状态
            send_rst = sr(IP(dst=dst_ip) / TCP(dport=dst_port, flags="AR"),
                          timeout=1)  # 全连接 AR => ACK+RST
            print(f"端口{dst_port}Open")
        else:
            print(tcp_connect_scan_resp.getlayer(TCP).flags)
            print(f"端口{dst_port}Closed")
    else:
        print(f"端口{dst_port}Closed")

def scan_S(ip, port):
    syn_connect_scan_resp = sr1(IP(dst=ip) / TCP(dport=port, flags="S"), timeout=10)
    if syn_connect_scan_resp is None:
        print(f"端口{port}Closed")
    elif syn_connect_scan_resp.haslayer(TCP):
        print(syn_connect_scan_resp.getlayer(TCP).flags)
        if syn_connect_scan_resp.getlayer(TCP).flags == "SA":
            print(f"端口{port}Open")
        else:
            print(syn_connect_scan_resp.getlayer(TCP).flags)
            print(f"端口{port}Closed")
    else:
        print(f"端口{port}Closed")

def scan_F(ip, port):
    fin_connect_scan_resp = sr1(IP(dst=ip) / TCP(dport=port, flags="F"), timeout=10)
    if fin_connect_scan_resp is None:
        print(f"端口{port}Open|Filtered")
    elif fin_connect_scan_resp.haslayer(TCP):
        print(fin_connect_scan_resp.getlayer(TCP).flags)
        if fin_connect_scan_resp.getlayer(TCP).flags == "RA":
            print(f"端口{port}Closed")
    else:
        print(f"端口{port}Closed")

def scan_X(ip, port):
    xmas_connect_scan_resp = sr1(IP(dst=ip) / TCP(dport=port, flags='FPU'), timeout=10)
    if xmas_connect_scan_resp is None:
        print(f"端口{port}Open|Filtered")
    elif xmas_connect_scan_resp.haslayer(TCP):
        print(xmas_connect_scan_resp.getlayer(TCP).flags)
        if xmas_connect_scan_resp.getlayer(TCP).flags == "RA":
         print(f"端口{port}Closed")
    else:
        print(f"端口{port}Closed")

def scan_N(ip, port):
    Null_connect_scan_resp = sr1(IP(dst=ip) / TCP(dport=port, flags=''), timeout=10)
    if Null_connect_scan_resp is None:
        print(f"端口{port}Open|Filtered")
    elif Null_connect_scan_resp.haslayer(TCP):
        print(Null_connect_scan_resp.getlayer(TCP).flags)
        if Null_connect_scan_resp.getlayer(TCP).flags == "RA":
            print(f"端口{port}Closed")
    else:
        print(f"端口{port}Closed")


def ping(host):
    """
    ping一个主机
    :param host: 主机地址
    :return:
    """
    package = IP(dst=host) / ICMP() / b'Welcome to my world!'
    reply = sr1(package, timeout=2, verbose=False)
    if reply:
        print(f'主机{host}可达')
    else:
        print(f'主机{host}不可达')

if __name__ == '__main__':
    dest = 'fangzhiyang.cn'
    # dest = '172.17.0.1'
    port = [22,9999]
    # ping(dest)
    for i in port:
        print("TCP")
        scan_T(dest, i)

        scan_S(dest, i)
        print("FIN")
        scan_F(dest, i)
